    int id PK
    string name

    int parentId FK
  }

    int id PK

  Product {

Обозначения: PK — первичный ключ, FK — внешний ключ, UK — уникальное поле. Линия ||—o< означает связь «один-ко-многим», >o—o< — «многие-ко-многим». Таблицы LoyaltyTier и CatalogFile — справочные, без внешних связей.
